```
class Carre() {
	float x = 0.0;
	float y = 0.0;
	float s;
	color c;
	
	Carre(color _couleur, float size) {
		c = couleur;
		s = size;
	}
	
	void draw() {
		noStroke();
		fill(c);
		rect(x,y,s,s);
	}
	
}
```
